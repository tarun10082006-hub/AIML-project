def normalize(s):
    return re.sub(r'[^a-z0-9]', '', s.lower())

def find_match(cls, folder_list):
    if cls in folder_list: return cls
    nc = normalize(cls)
    for f in folder_list:
        if normalize(f) == nc: return f
    for f in folder_list:
        if nc in normalize(f) or normalize(f) in nc: return f
    return None
